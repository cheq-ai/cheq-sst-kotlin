storage

@JvmStatic
val storage: Storage

Storage provides read/write access to cookies, local data, and session data for automatic inclusion in all Sst.trackEvent calls.

Sst.configure must be called once before methods on this object are called.

import ai.cheq.sst.android.core.Config
import ai.cheq.sst.android.core.Event

fun main() { 
   //sampleStart 
   runBlocking {
            Sst.configure(Config("client_name")) { context }

            // Add a cookie
            Sst.storage.cookies.add("cookieKey1", "cookieValue1")

            // Get all cookie key-value pairs
            var keyValuePairs = Sst.storage.cookies.all()
            println(keyValuePairs.size) // 1

            // Get a cookie value
            val value = Sst.storage.cookies.get("cookieKey1")?.value
            println(value) // cookieValue1

            // Clear the cookies
            Sst.storage.cookies.clear()

            // Get all cookie key-value pairs
            keyValuePairs = Sst.storage.cookies.all()
            println(keyValuePairs.size) // 0
        } 
   //sampleEnd
}
